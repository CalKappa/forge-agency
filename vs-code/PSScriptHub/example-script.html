Get-SystemInfo.ps1

A comprehensive PowerShell script to gather detailed system information including hardware specs, installed software, network configuration, and performance metrics. Perfect for system administrators who need quick system diagnostics.

Easy System Administration Diagnostics Hardware
Downloads: 2,847 Last Updated: March 15, 2024 Author: Admin Team

What This Script Does

The Get-SystemInfo script provides a comprehensive overview of your Windows system by collecting and displaying crucial information about:

  • System Hardware: CPU, RAM, disk space, and motherboard details
  • Operating System: Version, build number, installation date, and uptime
  • Network Configuration: IP addresses, network adapters, and connectivity status
  • Installed Software: List of installed programs and Windows features
  • System Performance: Current resource usage and performance counters

Requirements

  • Windows PowerShell 5.1 or PowerShell 7+
  • Administrator privileges (recommended for full system access)
  • Execution policy set to allow script execution
  • WMI/CIM cmdlets available (default in Windows)

Usage Instructions

  1. Open PowerShell as Administrator
  2. Navigate to the script location or run directly
  3. Execute: .\Get-SystemInfo.ps1
  4. Review the generated system report
  5. Optional: Export results to file with -ExportPath parameter

Parameters

-ExportPath [string]

Optional. Specify a file path to export the system information to a text file.

-IncludeSoftware [switch]

Include detailed installed software information (may take longer to execute).

-Detailed [switch]

Show additional detailed information including drivers and services.

Get-SystemInfo.ps1
# Get-SystemInfo.ps1 # Comprehensive system information gathering script # Author: PowerShell Scripts Hub # Version: 2.1 # Last Updated: March 15, 2024 param( [string]$ExportPath, [switch]$IncludeSoftware, [switch]$Detailed ) function Get-SystemInfo { param( [string]$ExportPath, [switch]$IncludeSoftware, [switch]$Detailed ) # Initialize output array $Output = @() # System Information Header $Output += "=== SYSTEM INFORMATION REPORT ===" $Output += "Generated: $((Get-Date).ToString('yyyy-MM-dd HH:mm:ss'))" $Output += "" # Computer System Information $ComputerSystem = Get-CimInstance -ClassName Win32_ComputerSystem $Output += "--- COMPUTER SYSTEM ---" $Output += "Computer Name: $($ComputerSystem.Name)" $Output += "Domain: $($ComputerSystem.Domain)" $Output += "Manufacturer: $($ComputerSystem.Manufacturer)" $Output += "Model: $($ComputerSystem.Model)" $Output += "Total Physical Memory: $([math]::Round($ComputerSystem.TotalPhysicalMemory / 1GB, 2)) GB" $Output += "" # Operating System Information $OS = Get-CimInstance -ClassName Win32_OperatingSystem $Output += "--- OPERATING SYSTEM ---" $Output += "OS Name: $($OS.Caption)" $Output += "Version: $($OS.Version)" $Output += "Build Number: $($OS.BuildNumber)" $Output += "Architecture: $($OS.OSArchitecture)" $Output += "Install Date: $($OS.InstallDate)" $Output += "Last Boot: $($OS.LastBootUpTime)" $Uptime = (Get-Date) - $OS.LastBootUpTime $Output += "Uptime: $($Uptime.Days) days, $($Uptime.Hours) hours, $($Uptime.Minutes) minutes" $Output += "" # Processor Information $Processor = Get-CimInstance -ClassName Win32_Processor $Output += "--- PROCESSOR ---" $Output += "Name: $($Processor.Name)" $Output += "Manufacturer: $($Processor.Manufacturer)" $Output += "Cores: $($Processor.NumberOfCores)" $Output += "Logical Processors: $($Processor.NumberOfLogicalProcessors)" $Output += "Max Clock Speed: $($Processor.MaxClockSpeed) MHz" $Output += "" # Memory Information $Memory = Get-CimInstance -ClassName Win32_PhysicalMemory $Output += "--- MEMORY ---" foreach ($Stick in $Memory) { $Output += "Memory Stick: $([math]::Round($Stick.Capacity / 1GB, 2)) GB - $($Stick.Speed) MHz - $($Stick.Manufacturer)" } $Output += "" # Disk Information $Disks = Get-CimInstance -ClassName Win32_LogicalDisk -Filter "DriveType = 3" $Output += "--- DISK DRIVES ---" foreach ($Disk in $Disks) { $FreeSpace = [math]::Round($Disk.FreeSpace / 1GB, 2) $TotalSize = [math]::Round($Disk.Size / 1GB, 2) $PercentFree = [math]::Round(($FreeSpace / $TotalSize) * 100, 1) $Output += "Drive $($Disk.DeviceID) - Total: $TotalSize GB, Free: $FreeSpace GB ($PercentFree% free)" } $Output += "" # Network Adapter Information $NetworkAdapters = Get-NetAdapter | Where-Object { $_.Status -eq "Up" } $Output += "--- NETWORK ADAPTERS ---" foreach ($Adapter in $NetworkAdapters) { $IPConfig = Get-NetIPAddress -InterfaceIndex $Adapter.InterfaceIndex -AddressFamily IPv4 -ErrorAction SilentlyContinue $Output += "Adapter: $($Adapter.Name)" $Output += " Description: $($Adapter.InterfaceDescription)" $Output += " Link Speed: $($Adapter.LinkSpeed)" if ($IPConfig) { $Output += " IP Address: $($IPConfig.IPAddress)" } } $Output += "" # Include software information if requested if ($IncludeSoftware) { $Output += "--- INSTALLED SOFTWARE ---" $Software = Get-CimInstance -ClassName Win32_Product | Sort-Object Name foreach ($Program in $Software) { $Output += "$($Program.Name) - Version: $($Program.Version)" } $Output += "" } # Include detailed information if requested if ($Detailed) { # Services $RunningServices = Get-Service | Where-Object { $_.Status -eq "Running" } $Output += "--- RUNNING SERVICES ($($RunningServices.Count) total) ---" foreach ($Service in $RunningServices | Select-Object -First 10) { $Output += "$($Service.Name) - $($Service.DisplayName)" } $Output += "... and $($RunningServices.Count - 10) more services" $Output += "" # System Performance $CPU = Get-Counter "\Processor(_Total)\% Processor Time" -SampleInterval 1 -MaxSamples 3 $AvgCPU = [math]::Round(($CPU.CounterSamples | Measure-Object -Property CookedValue -Average).Average, 2) $Output += "--- PERFORMANCE ---" $Output += "Average CPU Usage: $AvgCPU%" $MemoryUsage = [math]::Round((($OS.TotalVisibleMemorySize - $OS.FreePhysicalMemory) / $OS.TotalVisibleMemorySize) * 100, 2) $Output += "Memory Usage: $MemoryUsage%" } # Output results if ($ExportPath) { $Output | Out-File -FilePath $ExportPath -Encoding UTF8 Write-Host "System information exported to: $ExportPath" -ForegroundColor Green } else { $Output } } # Execute the function with provided parameters Get-SystemInfo -ExportPath $ExportPath -IncludeSoftware:$IncludeSoftware -Detailed:$Detailed
